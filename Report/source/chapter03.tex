\chapter{Methods}%
\label{ch:methods}%

This chapter formalizes the problem of learning a vision-based dexterous grasping policy and presents the mathematical framework underlying the two-stage privileged teacher--student pipeline.
\cref{sec:problem_formulation} states the problem as a partially observable decision process.
\cref{sec:privileged_teacher} derives the privileged teacher training objective.
\cref{sec:distillation_framework} introduces the distillation framework, covering DAgger, SafeDAgger, and DexSafeDagger.

\section{Problem Formulation}%
\label{sec:problem_formulation}%

\subsection{Dexterous Grasping as a POMDP}%
\label{sec:pomdp}%

The dexterous grasping task is modeled as a \ac{POMDP} defined by the tuple
%
\begin{equation}\label{eq:pomdp}
    \mathcal{M} = (\mathcal{S}, \mathcal{A}, \mathcal{O}, P, O, R, \gamma),
\end{equation}
%
where $\mathcal{S}$ is the state space, $\mathcal{A}$ the action space, $\mathcal{O}$ the observation space, $P: \mathcal{S} \times \mathcal{A} \to \Delta(\mathcal{S})$ the transition function, $O: \mathcal{S} \to \Delta(\mathcal{O})$ the observation function, $R: \mathcal{S} \times \mathcal{A} \to \mathbb{R}$ the reward function, and $\gamma \in [0,1)$ the discount factor.

The full state $s_t \in \mathcal{S}$ includes the joint configuration and velocities of the arm--hand system, the object pose, velocity, and physical properties, as well as contact information.
At deployment, the agent does not have access to $s_t$ directly; instead, it receives an observation $o_t \sim O(\cdot | s_t)$ consisting of proprioceptive measurements and camera images.
The goal is to find a policy $\pi: \mathcal{O} \to \Delta(\mathcal{A})$ that maximizes the expected discounted return
%
\begin{equation}\label{eq:objective}
    J(\pi) = \mathbb{E}_{\pi}\left[\sum_{t=0}^{T} \gamma^t R(s_t, a_t)\right].
\end{equation}

\subsection{Privileged vs.\ Partial Observability}%
\label{sec:priv_vs_partial}%

Learning directly under partial observability is difficult: the policy must simultaneously learn \emph{what} to do (the manipulation skill) and \emph{how} to perceive (extracting task-relevant information from raw sensor data).
The privileged teacher--student paradigm decouples these two challenges by introducing two distinct observation functions:
%
\begin{align}
    o_t^{\text{teacher}} &= O^{\text{priv}}(s_t), \label{eq:teacher_obs} \\
    o_t^{\text{student}} &= O^{\text{partial}}(s_t), \label{eq:student_obs}
\end{align}
%
where $O^{\text{priv}}$ is a deterministic, information-complete mapping that provides the teacher with the full state (joint states, object pose, contact forces), and $O^{\text{partial}}$ is the realistic observation function that provides the student with only proprioceptive measurements and RGB images.

The two-stage pipeline is then:
\begin{enumerate}
    \item \textbf{Stage~1:} Train a privileged teacher $\pi^*$ that maps $o_t^{\text{teacher}} \mapsto a_t$ by solving the \ac{MDP} induced by $O^{\text{priv}}$.
    \item \textbf{Stage~2:} Distill $\pi^*$ into a student $\hat{\pi}$ that maps $o_t^{\text{student}} \mapsto a_t$ via imitation learning.
\end{enumerate}

\section{Privileged Teacher Training}%
\label{sec:privileged_teacher}%

Since the teacher receives the full state, its training reduces to an \ac{MDP} $(\mathcal{S}, \mathcal{A}, P, R, \gamma)$.
The teacher is trained via \ac{PPO} with an asymmetric actor--critic architecture.

\subsection{Asymmetric Actor--Critic}%
\label{sec:asymmetric_ac}%

The actor (policy) and critic (value function) receive different observation vectors.
Let $o_t^{\text{actor}}$ denote the teacher's actor observation and $o_t^{\text{critic}}$ the critic observation, where $o_t^{\text{critic}}$ may contain additional privileged information (e.g., object velocities, contact forces) beyond $o_t^{\text{actor}}$.
The actor parameterizes a Gaussian policy
%
\begin{equation}\label{eq:actor}
    \pi_\theta(a_t | o_t^{\text{actor}}) = \mathcal{N}\left(\mu_\theta(o_t^{\text{actor}}),\; \sigma_\theta^2\right),
\end{equation}
%
while the critic estimates the state--value function
%
\begin{equation}\label{eq:critic}
    V_\phi(o_t^{\text{critic}}) \approx \mathbb{E}_\pi\left[\sum_{k=0}^{T-t} \gamma^k R(s_{t+k}, a_{t+k}) \;\middle|\; s_t\right].
\end{equation}
%
This asymmetry allows the critic to use richer information for more accurate value estimates during training, while the actor learns a policy that depends only on the subset of state available at deployment (before distillation to the student).

\subsection{PPO Objective}%
\label{sec:ppo_objective}%

The teacher is optimized using the clipped PPO surrogate objective~\cite{Schulman2017}:
%
\begin{equation}\label{eq:ppo_loss}
    L^{\text{CLIP}}(\theta) = \mathbb{E}_t\left[\min\left(r_t(\theta)\hat{A}_t,\;\text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon)\hat{A}_t\right)\right],
\end{equation}
%
where $r_t(\theta) = \pi_\theta(a_t | o_t^{\text{actor}}) / \pi_{\theta_{\text{old}}}(a_t | o_t^{\text{actor}})$ is the importance sampling ratio, $\hat{A}_t$ is the generalized advantage estimate (GAE)~\cite{Schulman2017}, and $\epsilon$ is the clipping parameter.

The advantage is computed using the critic via
%
\begin{equation}\label{eq:gae}
    \hat{A}_t = \sum_{l=0}^{T-t} (\gamma \lambda)^l \delta_{t+l}, \quad \delta_t = R(s_t, a_t) + \gamma V_\phi(o_{t+1}^{\text{critic}}) - V_\phi(o_t^{\text{critic}}),
\end{equation}
%
where $\lambda \in [0,1]$ is the GAE parameter that trades off bias and variance.

\subsection{Reward Structure}%
\label{sec:reward_structure}%

The reward function decomposes as a weighted sum of $K$ terms:
%
\begin{equation}\label{eq:reward}
    R(s_t, a_t) = \sum_{i=1}^{K} w_i \cdot r_i(s_t, a_t),
\end{equation}
%
where each $r_i$ captures a different aspect of the grasping task and $w_i \in \mathbb{R}$ is its weight.
The terms can be broadly categorized into:
\begin{itemize}
    \item \textbf{Approach rewards} that encourage the hand to move toward the object,
    \item \textbf{Contact rewards} that incentivize stable multi-finger contact,
    \item \textbf{Lift rewards} that reward raising the object to a goal height,
    \item \textbf{Regularization penalties} that discourage jerky or excessive actions.
\end{itemize}
The specific reward terms and their weights are detailed in the experimental setup (\cref{sec:reward_implementation}).

\subsection{Adaptive Domain Randomization}%
\label{sec:adr_formulation}%

To produce a robust teacher policy that generalizes across varying physical conditions, the training environment applies \ac{ADR}~\cite{OpenAI2020}.
Let $\boldsymbol{\xi} \in \Xi$ denote the vector of randomizable simulation parameters (e.g., friction, mass, actuator gains).
ADR maintains an interval $[\xi_i^{\text{lo}}, \xi_i^{\text{hi}}]$ for each parameter $\xi_i$ and uniformly samples $\xi_i \sim \mathcal{U}(\xi_i^{\text{lo}}, \xi_i^{\text{hi}})$ at each episode reset.

The intervals are widened according to a curriculum: when the policy's rolling success rate exceeds a threshold $\tau_{\text{ADR}}$, a shared increment counter $n$ is advanced, and each interval is linearly interpolated from its initial range toward its maximum range:
%
\begin{equation}\label{eq:adr}
    \xi_i^{\text{lo/hi}}(n) = \xi_i^{\text{start, lo/hi}} + \frac{n}{N_{\text{max}}} \left(\xi_i^{\text{max, lo/hi}} - \xi_i^{\text{start, lo/hi}}\right),
\end{equation}
%
where $N_{\text{max}}$ is the total number of ADR increments.
This creates a natural curriculum: early in training the policy sees near-nominal physics, and as it improves, the randomization range widens to cover progressively more challenging conditions.

\section{Policy Distillation}%
\label{sec:distillation_framework}%

Given a trained teacher $\pi^*$ operating on privileged observations, the goal of distillation is to train a student policy $\hat{\pi}_\theta: \mathcal{O}^{\text{partial}} \to \mathcal{A}$ that imitates the teacher using only partial observations.
The student minimizes the imitation loss
%
\begin{equation}\label{eq:imitation_loss}
    \mathcal{L}(\theta) = \mathbb{E}_{o_t^{\text{student}} \sim d_{\hat{\pi}}}\left[\left\| \hat{\pi}_\theta(o_t^{\text{student}}) - \pi^*(o_t^{\text{teacher}}) \right\|_2^2\right],
\end{equation}
%
where $d_{\hat{\pi}}$ is the state distribution induced by the student's own policy.
When both teacher and student output Gaussian distributions, an alternative is the KL divergence loss:
%
\begin{equation}\label{eq:kl_loss}
    \mathcal{L}_{\text{KL}}(\theta) = \mathbb{E}_{o_t^{\text{student}} \sim d_{\hat{\pi}}}\left[D_{\text{KL}}\!\left(\pi^*(\cdot | o_t^{\text{teacher}}) \;\|\; \hat{\pi}_\theta(\cdot | o_t^{\text{student}})\right)\right],
\end{equation}
%
which matches the full action distribution rather than only the mean, and has been found to outperform L2 regression in prior work on visuomotor distillation~\cite{Singh2025}.
The key challenge is that the training distribution $d_{\hat{\pi}}$ shifts as the student policy changes, causing compounding errors if naive behavioral cloning is used.

\subsection{DAgger}%
\label{sec:dagger_method}%

DAgger~\cite{Ross2011} addresses the distribution shift by iteratively collecting data under the student's own policy while labeling it with the teacher's actions.
The procedure is formalized in \cref{alg:dagger}.

\begin{algorithm}[htb]
\caption{DAgger~\cite{Ross2011}}
\label{alg:dagger}
\begin{algorithmic}[1]
\State Initialize dataset $\mathcal{D} \leftarrow \emptyset$, student policy $\hat{\pi}_{\theta_1}$
\For{$i = 1, \ldots, N$}
    \State Set roll-in policy $\pi_i = \beta_i \pi^* + (1 - \beta_i) \hat{\pi}_{\theta_i}$
    \State Collect trajectories $\{(o_t^{\text{student}}, s_t)\}$ by executing $\pi_i$
    \State Label: $a_t^* = \pi^*(o_t^{\text{teacher}})$ for each visited state
    \State Aggregate: $\mathcal{D} \leftarrow \mathcal{D} \cup \{(o_t^{\text{student}}, a_t^*)\}$
    \State Train $\hat{\pi}_{\theta_{i+1}}$ on $\mathcal{D}$ by minimizing \cref{eq:imitation_loss}
\EndFor
\State \Return $\hat{\pi}_{\theta_{N+1}}$
\end{algorithmic}
\end{algorithm}

The mixing coefficient $\beta_i \in [0, 1]$ controls the fraction of teacher vs.\ student rollouts.
In the simplest schedule, $\beta_1 = 1$ (pure teacher) and $\beta_i = 0$ for $i > 1$ (pure student).
After $N$ iterations, DAgger guarantees a policy with loss bounded by $\epsilon_N + O(1/N)$, where $\epsilon_N$ is the best achievable loss on $\mathcal{D}$.

\subsection{SafeDAgger}%
\label{sec:safedagger_method}%

While DAgger resolves distribution shift, it does not prevent the student from visiting dangerous states during data collection, nor does it prioritize teacher queries for states where the student is most likely to fail.
SafeDAgger~\cite{Zhang2016} augments DAgger with a learned \emph{safety policy} that determines when the teacher must intervene.

\subsubsection{Safety Policy}%
\label{sec:safety_policy}%

A binary classifier $D_\psi: \mathcal{O}^{\text{partial}} \to [0, 1]$ is trained to predict whether the student's action at a given observation is safe.
The safety label for each data point is defined as
%
\begin{equation}\label{eq:safety_label}
    y_t = \begin{cases}
        1 \quad (\text{safe}) & \text{if } \left\| \hat{\pi}_\theta(o_t^{\text{student}}) - \pi^*(o_t^{\text{teacher}}) \right\|_2 \leq \delta, \\
        0 \quad (\text{unsafe}) & \text{otherwise},
    \end{cases}
\end{equation}
%
where $\delta > 0$ is a threshold on the action disagreement between student and teacher.
The classifier is trained via binary cross-entropy on the aggregated dataset:
%
\begin{equation}\label{eq:safety_loss}
    \mathcal{L}_D(\psi) = -\mathbb{E}\left[y_t \log D_\psi(o_t) + (1 - y_t) \log(1 - D_\psi(o_t))\right].
\end{equation}

\subsubsection{Intervention Mechanism}%
\label{sec:intervention_mechanism}%

During data collection, the safety policy switches control between the student and teacher at each time step:
%
\begin{equation}\label{eq:safedagger_intervention}
    a_t = \begin{cases}
        \hat{\pi}_\theta(o_t^{\text{student}}) & \text{if } D_\psi(o_t^{\text{student}}) \geq 0.5, \\
        \pi^*(o_t^{\text{teacher}}) & \text{otherwise}.
    \end{cases}
\end{equation}
%
This creates an adaptive, state-dependent mixing policy: the teacher intervenes precisely when the student is predicted to deviate significantly, while the student acts autonomously in states where it is competent.
The resulting intervention rate
%
\begin{equation}\label{eq:beta_safedagger}
    \beta_{\text{safe}} = \frac{1}{|\mathcal{B}|}\sum_{t \in \mathcal{B}} \mathbb{1}\left[D_\psi(o_t^{\text{student}}) < 0.5\right]
\end{equation}
%
naturally decays as the student improves, providing an implicit curriculum from teacher-guided exploration to student-autonomous execution.
The complete SafeDAgger procedure is given in \cref{alg:safedagger}.

\begin{algorithm}[htb]
\caption{SafeDAgger~\cite{Zhang2016}. \hlsafe{Blue lines} indicate modifications relative to DAgger (\cref{alg:dagger}).}
\label{alg:safedagger}
\begin{algorithmic}[1]
\State Initialize dataset $\mathcal{D} \leftarrow \emptyset$, student $\hat{\pi}_{\theta_1}$, \hlsafe{safety policy $D_{\psi_1}$}
\For{$i = 1, \ldots, N$}
    \For{each time step $t$ in rollout}
        \If{\hlsafe{$D_{\psi_i}(o_t^{\text{student}}) \geq 0.5$}}
            \State \hlsafe{Execute $a_t = \hat{\pi}_{\theta_i}(o_t^{\text{student}})$} \Comment{\hlsafe{Student acts}}
        \Else
            \State \hlsafe{Execute $a_t = \pi^*(o_t^{\text{teacher}})$} \Comment{\hlsafe{Teacher intervenes}}
        \EndIf
        \State Query teacher: $a_t^* = \pi^*(o_t^{\text{teacher}})$
        \State \hlsafe{Compute safety label $y_t$ via \cref{eq:safety_label}}
        \State $\mathcal{D} \leftarrow \mathcal{D} \cup \{(o_t^{\text{student}}, a_t^*$, \hlsafe{$y_t$}$)\}$
    \EndFor
    \State Update $\hat{\pi}_{\theta_{i+1}}$ by minimizing \cref{eq:imitation_loss} on $\mathcal{D}$
    \State \hlsafe{Update $D_{\psi_{i+1}}$ by minimizing \cref{eq:safety_loss} on $\mathcal{D}$}
\EndFor
\State \Return $\hat{\pi}_{\theta_{N+1}}$
\end{algorithmic}
\end{algorithm}

\subsubsection{DexSafeDagger: Q-Value Risk Prediction}%
\label{sec:dexsafedagger_method}%

Zhang~\cite{ChiZhang2026} proposed an extension of SafeDAgger for dexterous manipulation that replaces the action-disagreement classifier with a \emph{Q-value-based risk predictor}.
This work, referred to as DexSafeDagger, has been submitted for publication but has not yet undergone peer review.
Rather than relying solely on the L2 distance between student and teacher actions, DexSafeDagger uses the teacher's learned value function $V_\phi$ to estimate the risk of the current state.

The intervention criterion becomes
%
\begin{equation}\label{eq:dexsafedagger}
    a_t = \begin{cases}
        \hat{\pi}_\theta(o_t^{\text{student}}) & \text{if } V_\phi(s_t) \geq V_{\text{safe}} \;\text{and}\; \|e_t\|_2 \leq \delta, \\
        \pi^*(o_t^{\text{teacher}}) & \text{otherwise},
    \end{cases}
\end{equation}
%
where $e_t = \hat{\pi}_\theta(o_t^{\text{student}}) - \pi^*(o_t^{\text{teacher}})$ is the action error, $V_{\text{safe}}$ is a value threshold, and $\delta$ is the action disagreement threshold.
The dual criterion captures two complementary failure modes: (1)~states where the student's action deviates from the teacher (behavioral error), and (2)~states that the teacher's own value function identifies as high-risk (environmental danger).

%
